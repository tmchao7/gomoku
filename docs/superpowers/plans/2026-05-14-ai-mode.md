```cpp
#ifndef GOMOKU_AI_H
#define GOMOKU_AI_H

#include "Gomoku/Board.h"
#include "Gomoku/Types.h"

namespace gomoku {

class AI {
public:
    enum class Difficulty { Easy, Hard };

    AI(Difficulty difficulty, Stone aiStone = Stone::White);

    /// 给定当前棋盘，返回 AI 计算出的最优落子位置
    Position bestMove(const Board& board) const;

    Difficulty difficulty() const;
    Stone aiStone() const;

private:
    Difficulty difficulty_;
    Stone aiStone_;

    /// 简单难度：遍历所有空位，对每个位置打分，取最高分
    Position evaluateBestMove(const Board& board) const;

    /// 困难难度：Minimax + α-β 剪枝，搜索深度 = 2
    Position minimaxBestMove(const Board& board) const;

    /// 对单个空位打分（从 stone 的视角），用于启发式评估
    int scorePosition(const Board& board, int row, int col, Stone stone) const;

    /// 全局面评估：对所有空位按己方和对方视角分别打分后加权求和
    int evaluate(const Board& board, Stone perspective) const;

    /// Minimax 递归搜索，maximizing = true 时是 AI 层（取最大值）
    int minimax(Board& board, int depth, int alpha, int beta, bool maximizing) const;
};

} // namespace gomoku

#endif // GOMOKU_AI_H
```

- [ ] **Step 3: 实现 scorePosition**

对一个空位打分。从 stone 的视角，假设该位置落子后，检查四个方向上形成的连续同色段：

```cpp
int AI::scorePosition(const Board& board, int row, int col, Stone stone) const {
    // 四个方向向量
    const int directions[4][2] = {{0, 1}, {1, 0}, {1, 1}, {1, -1}};
    int totalScore = 0;

    for (const auto& dir : directions) {
        // 正方向连续同色子数
        int positive = 0;
        int r = row + dir[0];
        int c = col + dir[1];
        while (board.isInside(r, c) && board.at(r, c) == stone) {
            ++positive;
            r += dir[0];
            c += dir[1];
        }
        const bool blocked1 = !board.isInside(r, c) || board.at(r, c) != Stone::Empty;

        // 反方向连续同色子数
        int negative = 0;
        r = row - dir[0];
        c = col - dir[1];
        while (board.isInside(r, c) && board.at(r, c) == stone) {
            ++negative;
            r -= dir[0];
            c -= dir[1];
        }
        const bool blocked2 = !board.isInside(r, c) || board.at(r, c) != Stone::Empty;

        const int count = positive + negative;  // 不含自身，配合 5 子判定
        const int openEnds = (blocked1 ? 0 : 1) + (blocked2 ? 0 : 1);

        if (count >= 4) {
            totalScore += 100000;  // 连五 / 超五，最优先
        } else if (count == 3) {
            totalScore += openEnds == 0 ? 200 : (openEnds == 1 ? 5000 : 8000);
        } else if (count == 2) {
            totalScore += openEnds == 0 ? 30 : (openEnds == 1 ? 500 : 1000);
        } else if (count == 1) {
            totalScore += openEnds == 0 ? 10 : (openEnds == 1 ? 50 : 200);
        } else {
            totalScore += openEnds * 15;  // 孤子，开路多更容易发展
        }
    }

    // 星位奖励：中心及四个角星位
    const int stars[5][2] = {{3, 3}, {3, 11}, {7, 7}, {11, 3}, {11, 11}};
    for (const auto& star : stars) {
        const int dist = std::abs(row - star[0]) + std::abs(col - star[1]);
        if (dist <= 1) totalScore += 5;
    }

    return totalScore;
}
```

```cpp
int AI::evaluate(const Board& board, Stone perspective) const {
    const Stone opponent = oppositeStone(perspective);
    int score = 0;

    for (int r = 0; r < kBoardSize; ++r) {
        for (int c = 0; c < kBoardSize; ++c) {
            if (board.at(r, c) != Stone::Empty) continue;

            score += scorePosition(board, r, c, perspective);
            score += static_cast<int>(scorePosition(board, r, c, opponent) * 0.9);
        }
    }

    return score;
}
```

```cpp
Position AI::evaluateBestMove(const Board& board) const {
    const Stone me = aiStone_;
    const Stone opp = oppositeStone(me);

    Position bestPos{7, 7};  // 默认中心
    int bestScore = -1;

    for (int r = 0; r < kBoardSize; ++r) {
        for (int c = 0; c < kBoardSize; ++c) {
            if (board.at(r, c) != Stone::Empty) continue;

            const int attack = scorePosition(board, r, c, me);
            const int defend = scorePosition(board, r, c, opp);
            // 进攻分 + 防守分（对方在该位置的价值），防略低于攻
            const int total = attack + static_cast<int>(defend * 0.9);

            if (total > bestScore) {
                bestScore = total;
                bestPos = {r, c};
            }
        }
    }

    return bestPos;
}
```

- [ ] **Step 6: 实现 minimax 递归**

```cpp
int AI::minimax(Board& board, int depth, int alpha, int beta, bool maximizing) const {
    const Stone perspective = maximizing ? aiStone_ : oppositeStone(aiStone_);

    // 叶节点：评估当前局面
    if (depth == 0) {
        return evaluate(board, aiStone_);
    }

    // 检查是否有玩家已获胜
    for (int r = 0; r < kBoardSize; ++r) {
        for (int c = 0; c < kBoardSize; ++c) {
            if (board.at(r, c) != Stone::Empty && board.hasFiveInRow(r, c)) {
                const Stone winner = board.at(r, c);
                return winner == aiStone_ ? 1000000 : -1000000;
            }
        }
    }

    if (maximizing) {
        int maxScore = std::numeric_limits<int>::min();
        for (int r = 0; r < kBoardSize; ++r) {
            for (int c = 0; c < kBoardSize; ++c) {
                if (!board.isEmpty(r, c)) continue;

                board.placeStone(r, c, perspective);
                const int score = minimax(board, depth - 1, alpha, beta, false);
                board.removeStone(r, c);

                maxScore = std::max(maxScore, score);
                alpha = std::max(alpha, score);
                if (beta <= alpha) break;  // β 剪枝
            }
            if (beta <= alpha) break;
        }
        return maxScore;
    } else {
        int minScore = std::numeric_limits<int>::max();
        for (int r = 0; r < kBoardSize; ++r) {
            for (int c = 0; c < kBoardSize; ++c) {
                if (!board.isEmpty(r, c)) continue;

                board.placeStone(r, c, perspective);
                const int score = minimax(board, depth - 1, alpha, beta, true);
                board.removeStone(r, c);

                minScore = std::min(minScore, score);
                beta = std::min(beta, score);
                if (beta <= alpha) break;  // α 剪枝
            }
            if (beta <= alpha) break;
        }
        return minScore;
    }
}
```

```cpp
Position AI::minimaxBestMove(const Board& board) const {
    Board mutableBoard = board;
    Position bestPos{7, 7};
    int bestScore = std::numeric_limits<int>::min();

    for (int r = 0; r < kBoardSize; ++r) {
        for (int c = 0; c < kBoardSize; ++c) {
            if (!mutableBoard.isEmpty(r, c)) continue;

            mutableBoard.placeStone(r, c, aiStone_);
            const int score = minimax(mutableBoard, 2,  // depth = 2
                                      std::numeric_limits<int>::min(),
                                      std::numeric_limits<int>::max(),
                                      false);
            mutableBoard.removeStone(r, c);

            if (score > bestScore) {
                bestScore = score;
                bestPos = {r, c};
            }
        }
    }

    return bestPos;
}
```

```cpp
void BoardWidget::executeAIMove() {
    // 二次检查：可能在延迟期间游戏已结束
    if (gameOver_) {
        updateControlButtons();
        update();
        return;
    }

    const gomoku::Position pos = ai_->bestMove(board_);
    if (!board_.placeStone(pos.row, pos.col, ai_->aiStone())) {
        // AI 计算出非法位置，回退到安全处理
        updateControlButtons();
        update();
        return;
    }

    replay_.addMove({pos.row, pos.col, ai_->aiStone(),
                     playerName(ai_->aiStone()).toStdString()});

    update();

    if (board_.hasFiveInRow(pos.row, pos.col)) {
        gameOver_ = true;
        updateControlButtons();
        update();
        showWinner(ai_->aiStone());
        return;
    }

    if (board_.isFull()) {
        gameOver_ = true;
        updateControlButtons();
        update();
        QMessageBox::information(this, "游戏结束", "棋盘已满，平局！");
        return;
    }

    switchPlayer();
}
```
